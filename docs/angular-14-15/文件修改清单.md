## 🛠️ 批量修改脚本

### 方式一：使用 sed

```bash
#!/bin/bash
# rxjs7-migrate.sh

# 1. throwError 修改
find src -name "*.ts" -type f ! -name "*.spec.ts" -exec sed -i 's/throwError(\([^()]*\))/throwError(() => \1)/g' {} +

# 2. Observable.forkJoin
find src -name "*.ts" -type f -exec sed -i 's/Observable\.forkJoin/forkJoin/g' {} +

# 3. Observable.of (仅测试文件)
find src -name "*.spec.ts" -type f -exec sed -i 's/Observable\.of(/of(/g' {} +

# 4. Observable.fromEvent
find src -name "*.ts" -type f -exec sed -i 's/Observable\.fromEvent/fromEvent/g' {} +

# 5. Observable.combineLatest
find src -name "*.ts" -type f -exec sed -i 's/Observable\.combineLatest/combineLatest/g' {} +

# 6. .do() → .pipe(tap()) - 需要手动处理
echo "⚠️  .do() 操作符需要手动修改为 .pipe(tap())"

echo "✅ 批量替换完成，请检查导入语句并手动修复 .do() 操作符"
```

### 方式二：使用 TypeScript Compiler API

```typescript
// scripts/rxjs7-migrate.ts
import * as ts from 'typescript';
import * as fs from 'fs';
import * as path from 'path';

function migrateFile(filePath: string) {
  const sourceCode = fs.readFileSync(filePath, 'utf-8');
  const sourceFile = ts.createSourceFile(
    filePath,
    sourceCode,
    ts.ScriptTarget.Latest,
    true
  );

  // 使用 TypeScript Compiler API 进行精确的 AST 转换
  // ... 实现代码

  console.log(`Migrated: ${filePath}`);
}

function migrateDirectory(dir: string) {
  const files = fs.readdirSync(dir);
  files.forEach(file => {
    const filePath = path.join(dir, file);
    const stat = fs.statSync(filePath);
    if (stat.isDirectory()) {
      migrateDirectory(filePath);
    } else if (file.endsWith('.ts') && !file.endsWith('.spec.ts')) {
      migrateFile(filePath);
    }
  });
}

migrateDirectory('src');
```
